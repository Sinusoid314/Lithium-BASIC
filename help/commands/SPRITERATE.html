SPRITERATE winName, spriteName, rate


Description:

Sets the animation rate of the sprite spriteName on the window or control winName. The higher the rate number is, the slower the sprite's frame cycle will be. For example, a rate of 50 will speed up the sprite's animation, and a rate of 500 will slow it down.


Usage:

loadimg "img1", DefPath & "\examples\sprite1-1.bmp"
loadimg "img2", DefPath & "\examples\sprite1-2.bmp"

window "win", "Sprite Test", normal, 100, 100, 500, 500
control "draw1", "win", "", drawbox, 10, 10, 400, 300
control "btnRate", "win", "Set Rate", button, 200, 365, 100, 25

event "win", "close", Close
event "btnRate", "click", SetRate

sprite "draw1", "sprite1", 10, 10, "img1"
addframe "draw1", "sprite1", "img2"

timer "timer1", 100, RedrawSprites

pause

sub RedrawSprites
     drawsprites "draw1"
end sub

sub SetRate
     var rate as number
     rate = val(inputbox("Enter sprite rate:", "Rate"))
     SpriteRate "draw1", "sprite1", rate
end sub

sub Close
     unloadimg "img1"
     unloadimg "img2"
     end
end sub
