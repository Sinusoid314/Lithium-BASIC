SPRITEPLAY winName, spriteName


Description:

Causes the sprite spriteName on the window or control winName to resume cycling of its image frames. The sprite's animation can be paused by using the SPRITEPAUSE command.


Usage:

loadimg "img1", DefPath & "\examples\sprite1-1.bmp"
loadimg "img2", DefPath & "\examples\sprite1-2.bmp"

window "win", "Sprite Test", normal, 100, 100, 500, 400
control "draw1", "win", "", drawbox, 10, 10, 140, 100
control "btnPlay", "win", "Play", button, 160, 10, 100, 25
control "btnPause", "win", "Pause", button, 160, 40, 100, 25

event "win", "close", Close
event "btnPause", "click", PauseSprite
event "btnPlay", "click", PlaySprite

sprite "draw1", "sprite1", 10, 10, "img1"
addframe "draw1", "sprite1", "img2"

timer "timer1", 100, RedrawSprites

pause

sub RedrawSprites
     drawsprites "draw1"
end sub

sub PauseSprite
     spritepause "draw1", "sprite1"
end sub

sub PlaySprite
     SpritePlay "draw1", "sprite1"
end sub

sub Close
     unloadimg "img1"
     unloadimg "img2"
     end
end sub
