SPRITESIZE winName, spriteName, width, height


Description:

Sets the width and height of the sprite specified by spriteName on the window or control winName.


Usage:

loadimg "img1", DefPath & "\examples\sprite1-1.bmp"
loadimg "img2", DefPath & "\examples\sprite1-2.bmp"

window "win", "Sprite Test", normal, 100, 100, 500, 500
control "draw1", "win", "", drawbox, 10, 10, 400, 300
control "txtWidth", "win", "100", textbox, 160, 350, 100, 20
control "txtHeight", "win", "30", textbox, 160, 380, 100, 20
control "btnSize", "win", "Resize", button, 280, 365, 100, 25

event "win", "close", Close
event "btnSize", "click", ResizeSprite

sprite "draw1", "sprite1", 10, 10, "img1"
addframe "draw1", "sprite1", "img2"

timer "timer1", 100, RedrawSprites

pause

sub RedrawSprites
     drawsprites "draw1"
end sub

sub ResizeSprite
     var w, h as number
     w = val(gettext("txtWidth"))
     h = val(gettext("txtHeight"))
     SpriteSize "draw1", "sprite1", w, h
end sub

sub Close
     unloadimg "img1"
     unloadimg "img2"
     end
end sub
