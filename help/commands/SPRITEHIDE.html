SPRITEHIDE winName, spriteName


Description:

Hides the sprite specified by spriteName on the window or control winName. The sprite can be shown by using the SPRITESHOW command.


Usage:

loadimg "img1", DefPath & "\examples\sprite1-1.bmp"
loadimg "img2", DefPath & "\examples\sprite1-2.bmp"

window "win", "Sprite Test", normal, 100, 100, 500, 400
control "draw1", "win", "", drawbox, 10, 10, 140, 100
control "btnShow", "win", "Show", button, 160, 10, 100, 25
control "btnHide", "win", "Hide", button, 160, 40, 100, 25

event "win", "close", Close
event "btnShow", "click", ShowSprite
event "btnHide", "click", HideSprite

sprite "draw1", "sprite1", 10, 10, "img1"
addframe "draw1", "sprite1", "img2"

timer "timer1", 100, RedrawSprites

pause

sub RedrawSprites
     drawsprites "draw1"
end sub

sub ShowSprite
     spriteshow "draw1", "sprite1"
end sub

sub HideSprite
     SpriteHide "draw1", "sprite1"
end sub

sub Close
     unloadimg "img1"
     unloadimg "img2"
     end
end sub
