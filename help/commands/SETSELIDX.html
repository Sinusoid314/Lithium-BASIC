SETSELIDX name, index


Description:

Selects an item, specified by index, in the listbox or combobox specified by name.


Usage:

window "win", "Select Item Test", normal, 100, 100, 500, 400
control "list1", "win", "", listbox, 100, 20, 100, 200
control "btn1", "win", "Select", button, 250, 50, 100, 25

additem "list1", "Item 1"
additem "list1", "Item 2"
additem "list1", "Item 3"

event "btn1", "click", Btn1Click

pause

sub Btn1Click
     var idx as number
     idx = val(inputbox("Enter the index of an item:", "Select Item"))
     SetSelIdx "list1", idx
end sub
