SPRITEROTATE winName, spriteName, rotation


Description:

Rotates the sprite spriteName on the window or control winName. The rotation is specified by rotation, which is a string that can hold one of the four rotations values: normal, flip, mirror, and rotate180.


Usage:

loadimg "img1", DefPath & "\examples\sprite1-1.bmp"
loadimg "img2", DefPath & "\examples\sprite1-2.bmp"

window "win", "Sprite Test", normal, 100, 100, 500, 500
control "draw1", "win", "", drawbox, 10, 10, 400, 300
control "btnNormal", "win", "Normal", button, 150, 365, 100, 25
control "btnFlip", "win", "Flip", button, 260, 365, 100, 25
control "btnMirror", "win", "Mirror", button, 150, 400, 100, 25
control "btnRotate", "win", "Rotate180", button, 260, 400, 100, 25

event "win", "close", Close
event "btnNormal", "click", Normal
event "btnFlip", "click", Flip
event "btnMirror", "click", Mirror
event "btnRotate", "click", Rotate180

sprite "draw1", "sprite1", 100, 100, "img1"
addframe "draw1", "sprite1", "img2"

timer "timer1", 100, RedrawSprites

pause

sub RedrawSprites
     drawsprites "draw1"
end sub

sub Normal
     SpriteRotate "draw1", "sprite1", "normal"
end sub

sub Flip
     SpriteRotate "draw1", "sprite1", "flip"
end sub

sub Mirror
     SpriteRotate "draw1", "sprite1", "mirror"
end sub

sub Rotate180
     SpriteRotate "draw1", "sprite1", "rotate180"
end sub

sub Close
     unloadimg "img1"
     unloadimg "img2"
     end
end sub
