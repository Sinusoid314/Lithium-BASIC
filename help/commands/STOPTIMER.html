STOPTIMER name


Description:

Stops and destroys the timer specified by name. The timer must have already been created using the TIMER command.


Usage:

var count as number

window "win", "Timer Test", normal, 100, 100, 500, 400
control "countTxt", "win", "0", statictext, 200, 50, 100, 20
control "startBtn", "win", "Start Timer", button, 200, 100, 100, 25
control "stopBtn", "win", "Stop Timer", button, 200, 200, 100, 25

disable "startBtn"

timer "counter", 1000, CountSub

event "startBtn", "click", StartCount
event "stopBtn", "click", StopCount

pause

sub CountSub
     count = count + 1
     settext "countTxt", count
end sub

sub StopCount
     StopTimer "counter"
     disable "stopBtn"
     enable "startBtn"
end sub

sub StartCount
     timer "counter", 1000, CountSub
     disable "startBtn"
     enable "stopBtn"
end sub
