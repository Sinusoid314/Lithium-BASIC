SETITEM name, index, text


Description:

Sets the text of an item in a listbox or combobox specified by name. The item is specified by index.


Usage:

window "win", "Set Item Test", normal, 100, 100, 500, 400
control "list1", "win", "", listbox, 100, 20, 100, 200
control "btn1", "win", "Set Item", button, 250, 50, 100, 25

additem "list1", "Item 1"
additem "list1", "Item 2"
additem "list1", "Item 3"

event "btn1", "click", Btn1Click

pause

sub Btn1Click
     var newText as string
     newText = inputbox("Enter new item text:", "Set Item")
     SetItem "list1", 2, newText
end sub
