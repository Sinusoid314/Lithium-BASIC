SPRITEPOS winName, spriteName, x, y


Description:

Sets the x and y position of the sprite specified by spriteName on the window or control winName.


Usage:

loadimg "img1", DefPath & "\examples\sprite1-1.bmp"
loadimg "img2", DefPath & "\examples\sprite1-2.bmp"

window "win", "Sprite Test", normal, 100, 100, 500, 500
control "draw1", "win", "", drawbox, 10, 10, 400, 300
control "txtLeft", "win", "10", textbox, 160, 350, 100, 20
control "txtTop", "win", "10", textbox, 160, 380, 100, 20
control "btnMove", "win", "Move", button, 280, 365, 100, 25

event "win", "close", Close
event "btnMove", "click", MoveSprite

sprite "draw1", "sprite1", 10, 10, "img1"
addframe "draw1", "sprite1", "img2"

timer "timer1", 100, RedrawSprites

pause

sub RedrawSprites
     drawsprites "draw1"
end sub

sub MoveSprite
     var x, y as number
     x = val(gettext("txtLeft"))
     y = val(gettext("txtTop"))
     SpritePos "draw1", "sprite1", x, y
end sub

sub Close
     unloadimg "img1"
     unloadimg "img2"
     end
end sub
