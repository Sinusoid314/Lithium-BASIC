GETSOUNDPOS(name)


Description:

Returns the current position of the sound specified by name.


Usage:

var musicLen as number
var v as string

showconsol

opensound "mid", DefPath & "\examples\the guess who - no time.mid"
playsound "mid"

timer "midTimer", 200, CheckDone

musicLen = getsoundlen("mid")
input "Press enter to quit ", v
closesound "mid"

sub CheckDone
     var musicPos as number
     musicPos = GetSoundPos("mid")
     if musicPos = musicLen then
          closesound "mid"
          end
     end if
end sub
